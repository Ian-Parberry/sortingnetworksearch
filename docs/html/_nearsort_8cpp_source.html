55bool CNearsort::nearsorts(){
56 m_pGrayCode->initalize();
57 initValues(1, DEPTH - 3);
58
59 for(int i=0; i<INPUTS; i++){
60 for(int j=0; j<INPUTS; j++)
61 m_bReachableFrom[i][j] = m_bReachableTo[i][j] = m_bReachable[i][j] = false;
62
63 m_nReachCountFrom[i] = m_nReachCountTo[i] = m_nReachCount[i] = 0;
64 } //for
65
66 if(!evennearsorts())
67 return false;
68
69 //if odd number of inputs, handle the last one
70
71 if(odd(INPUTS)){
72 m_pGrayCode->initalize();
73 initValues(1, DEPTH - 3);
74
75 for(int j=1; j<DEPTH; j++)
76 m_nValue[j][INPUTS - 1] = 1;
77
78 m_pGrayCode->m_nZeros = INPUTS - 1;
79
80 if(!evennearsorts())
81 return false;
82 } //if
83
84 return true;
85} //nearsorts

92bool CNearsort::stillnearsorts(const size_t delta){
93 size_t j = flipinput(delta - 1, 1, DEPTH - 3); //source channel into level d-2
94 const size_t k = m_pGrayCode->m_nZeros +
95 m_pGrayCode->m_nBit[delta] - 1; //target channel out of level d-1
96
97 if(j == k)return true; //self
98
99 //reachability heuristic: size of "from" <= 3
100
101 if(!m_bReachableFrom[j][k]){
102 if(m_nReachCountFrom[j] >= 3)return false; //not there and no room
103 m_nReachCountFrom[j]++;
104 m_bReachableFrom[j][k] = true;
105 } //if
106
107 //reachability heuristic: size of "to" <= 3
108
109 if(!m_bReachableTo[j][k]){
110 if(m_nReachCountTo[k] >= 3)return false; //not there and no room
111 m_nReachCountTo[k]++;
112 m_bReachableTo[j][k] = true;
113 } //if
114
115 //reachability heuristic: size of "from" union "to" <= 5
116 if(!m_bReachable[j][k]){
117 if(m_nReachCount[j] >= 5 || m_nReachCount[k] >= 5) return false; //not there and no room
118 m_nReachCount[j]++; m_nReachCount[k]++;
119 m_bReachable[j][k] = m_bReachable[k][j] = true;
120 } //if
121
122 return true;
123} //stillnearsorts
