72bool C1NF::EvenSorts(){
73 size_t i = 0; //index of bit to flip
74 bool bSorts = true; //assume it sorts until we find otherwise
75
76 while(bSorts && i < m_nWidth){ //bail if it doesn't sort, or we've tried all binary inputs
77 i = m_pGrayCode->Next(); //next bit to flip in Gray code order
78 bSorts = bSorts && (i >= m_nWidth || StillSorts(i)); //check whether it still sorts when this bit is flipped
79 } //while
80
81 return bSorts;
82} //EvenSorts

90bool C1NF::Sorts(){
91 //first handle the case where n is even, and the case where n is odd
92 //and fails to sort an input that ends with a zero
93 Initialize(); //set all channels to zero
94 if(!EvenSorts())return false; //test inputs ending in zero
95
96 //if odd number of inputs, check input that end with a one
97 if(odd(m_nWidth)){
98 Initialize(); //set all channels to zero
99
100 for(int j=0; j<m_nDepth; j++) //set all values on last channel to one
101 m_nValue[j][m_nWidth - 1] = 1;
102
103 m_nZeros = m_nWidth - 1; //correct the count of zeros
104
105 if(!EvenSorts())return false; //test inputs ending with one
106 } //if
107
108 return true; //Oh, we made it this far? Then I must be a sorting network. Hurray!
109} //sorts
