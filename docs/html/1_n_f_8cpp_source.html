71bool C1NF::evensorts(){
72 size_t i = 0; //index of bit to flip
73 bool bSorts = true; //assume it sorts until we find otherwise
74
75 while(bSorts && i <= m_nWidth){ //bail if it doesn't sort, or we've tried all binary inputs
76 i = m_pGrayCode->next(); //next bit to flip in Gray code order
77 bSorts = bSorts && (i > m_nWidth || stillsorts(i)); //check whether it still sorts when this bit is flipped
78 } //while
79
80 return bSorts;
81} //evensorts

89bool C1NF::sorts(){
90 //first handle the case where n is even, and the case where n is odd
91 //and fails to sort an input that ends with a zero
92 initialize(); //set all channels to zero
93 if(!evensorts())return false; //test inputs ending in zero
94
95 //if odd number of inputs, check input that end with a one
96 if(odd(m_nWidth)){
97 initialize(); //set all channels to zero
98
99 for(int j=0; j<m_nDepth; j++) //set all values on last channel to one
100 m_nValue[j][m_nWidth - 1] = 1;
101
102 m_pGrayCode->SetNumZeros(m_nWidth - 1); //correct the count of zeros
103
104 if(!evensorts())return false; //test inputs ending with one
105 } //if
106
107 return true; //Oh, we made it this far? Then I must be a sorting network. Hurray!
108} //sorts
