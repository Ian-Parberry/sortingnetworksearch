44size_t CComparatorNetwork::RemoveRepeatedComparators(){
45 size_t count = 2*(m_nWidth/2); //number of comparators
46
47 for(size_t i=0; i<m_nDepth; i++) //for each level
48 for(size_t j=0; j<m_nWidth; j++) //for each channel
49 m_bRedundant[i][j] = false;
50
51 for(size_t i=1; i<m_nDepth; i++) //for each level except the top one
52 for(size_t j=0; j<m_nWidth; j++) //for each channel
53 if((m_nComparator[i][j] == m_nComparator[i-1][j]) && !m_bRedundant[i][j]) //if channel connected to same one in previous level
54 m_bRedundant[i][j] = true; //remove connection
55 else if(m_nComparator[i][j] != j)count++;
56
57 return count/2; //everything gets counted twice
58} //RemoveRepeatedComparators
