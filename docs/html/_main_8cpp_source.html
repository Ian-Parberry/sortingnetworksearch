164void Search(CThreadManager* p, const size_t nDepth, const bool bNearsort2){
165 CLevel2Search* pLevel2Search = new CLevel2Search(); //for level 2 matchings
166 auto L2Matchings = pLevel2Search->GetMatchings(); //get level 2 matchings
167 size_t i = 0; //index of current matching
168
169 //insert search tasks to task queue
170
171 for(auto matching: L2Matchings){ //for each level2 matching
172 CSearchable* pSearch = nullptr; //for the searchable sorting network
173
174 switch(nDepth){ //choose optimization depending on depth
175 case 2: pSearch = new C2NF(matching, i++); break;
176 case 3: pSearch = new CAutocomplete(matching, i++); break;
177 case 4: pSearch = new CNearsort(matching, i++); break;
178 default: //depth 5 or greater
179 if(bNearsort2)
180 pSearch = new CNearsort2(matching, i++);
181 else pSearch = new CNearsort(matching, i++);
182 break;
183 } //switch
184
185 p->Insert(new CTask(pSearch)); //insert search task
186 } //for
187
188 //perform multi-threaded backtracking search
189
190 p->Spawn(); //spawn threads
191 p->Wait(); //wait for threads to finish
192 p->Process(); //process results
193
194 delete pLevel2Search;
195} //Search
