39CLevel2Search::CLevel2Search(){
40 size_t m_nCurIndex = 0; //index of current matching - no need to call GetIndex
41
42 CMatching curMatching; //current matching
43
44 do{ //for each matching
45 if(odd(m_nWidth))curMatching[m_nWidth] = m_nWidth; //dummy last channel for odd width networks
46 CMatching copy(curMatching);
47
48 if(m_stlUsed.find(m_nCurIndex) == m_stlUsed.end()){ //if it is not used
49 const size_t minIndex = std::min(m_nCurIndex, Permute(copy, m_nWidth/2)); //all pairwise permutations used also
50
51 if(m_stlUsed.find(minIndex) == m_stlUsed.end()){ //if it is not used
52 m_stlResults.push_back(curMatching); //insert into results
53 m_stlUsed.insert(minIndex); //mark it used
54 } //if
55 } //if
56
57 m_nCurIndex++; //index of next matching
58 }while(curMatching.Next()); //get next matching, exit if there is none
59
60 //normalize the resulting matchings
61
62 for(auto m: m_stlResults)
63 m.Normalize();
64
65 //sort the result matchings by index (not necessary but useful when debugging)
66
67 //std::sort(m_stlResults.begin(), m_stlResults.end());
68 std::cout << m_stlResults.size() << " second levels found" << std::endl;
69 Save(); //save results to file
70} //constructor

77size_t CLevel2Search::GetIndex(CMatching& matching){
78 size_t index = 0; //the index value to be returned
79 size_t b = GetNumMatchings(m_nWidth)/(oddfloor(m_nWidth)); //block size to be skipped
80
81 int nCopy[MAXINPUTS + 1] = {0}; //integer copy of matching.
82
83 //grab a copy of matching into the array m
84
85 for(size_t i=0; i<=m_nWidth; i++)
86 nCopy[i] = (int)matching[i];
87
88 //make sure that each comparator is listed in ascending order
89
90 for(size_t i=0; i<m_nWidth; i+=2)
91 if(nCopy[i] > nCopy[i+1])
92 std::swap(nCopy[i], nCopy[i+1]);
93
94 //compute index
95
96 for(size_t n=evenceil(m_nWidth); n>2; n-=2){ //for each comparator
97 size_t max = 1; //the index of the largest channel on the max side of a comparator
98
99 for(size_t i=1; i<=m_nWidth; i+=2)
100 if(nCopy[i] > nCopy[max])
101 max = i;
102
103 //skip over the matchings that have max matching to something smaller
104 //than max is matching to in the current matching, which is m[max - 1]
105
106 index += b*(n - 2 - nCopy[max - 1]);
107
108 //remove m_nCopy[max - 1] from the matching
109
110 for(size_t j=0; j<m_nWidth; j++)
111 if(nCopy[j] > nCopy[max - 1])
112 nCopy[j]--;
113
114 //prepare for next iteration
115
116 b /= n - 3; //block size to be skipped is now smaller
117 nCopy[max] = 0; //remove channel max from consideration
118 } //for
119
120 return index;
121} //GetIndex

129size_t CLevel2Search::Permute(CMatching& matching, const size_t n){
130 size_t nMinIndex = 99999; //minimum matching index found
131
132 for(size_t i=0; i<n; i++){
133 if(n > 2) //recurse on sub-permutation
134 nMinIndex = std::min(nMinIndex, Permute(matching, n - 1));
135
136 if(i < n - 1){ //base of recursion
137 matching.Swap(odd(n)? 0: i, n - 1); //swap to new candidate
138 nMinIndex = std::min(nMinIndex, GetIndex(matching));
139 } //if
140 } //for
141
142 return nMinIndex;
143} //Permute
