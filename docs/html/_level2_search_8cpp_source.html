35CLevel2Search::CLevel2Search(){
36 //printf("Searching for level 2 candidates on %d inputs.\n", INPUTS);
37 size_t m_nCurIndex = 0; //index of current matching - no need to call GetIndex
38
39 CMatching curMatching; //current matching
40
41 do{ //for each matching
42 if(odd(INPUTS))curMatching[INPUTS] = INPUTS; //dummy last channel for odd width networks
43 CMatching copy(curMatching);
44
45 if(m_stlUsed.find(m_nCurIndex) == m_stlUsed.end()){ //if it is not used
46 const size_t minIndex = std::min(m_nCurIndex, Permute(copy, INPUTS/2)); //all pairwise permutations used also
47
48 if(m_stlUsed.find(minIndex) == m_stlUsed.end()){ //if it is not used
49 m_stlResults.push_back(curMatching); //insert into results
50 m_stlUsed.insert(minIndex); //mark it used
51 } //if
52 } //if
53
54 m_nCurIndex++; //index of next matching
55 }while(curMatching.Next()); //get next matching, exit if there is none
56
57 //normalize the resulting matchings
58
59 for(auto m: m_stlResults)
60 m.Normalize();
61
62 //sort the result matchings by index (not necessary but useful when debugging)
63
64 std::sort(m_stlResults.begin(), m_stlResults.end());
65} //constructor

89const size_t CLevel2Search::GetIndex(CMatching& matching) const{
90 size_t index = 0; //the index value to be returned
91 size_t b = GetNumMatchings(INPUTS)/(oddfloor(INPUTS)); //block size to be skipped
92 size_t m[INPUTS + 1]; //local copy of matching to manipulate
93
94 //grab a copy of matching into the array m
95
96 for(size_t i=0; i<=INPUTS; i++)
97 m[i] = matching[i];
98
99 //make sure that each comparator is listed in ascending order
100
101 for(size_t i=0; i<INPUTS; i+=2)
102 if(m[i] > m[i+1])
103 std::swap(m[i], m[i+1]);
104
105 //compute index
106
107 for(size_t n=evenceil(INPUTS); n>2; n-=2){ //for each comparator
108 size_t max = 1; //the index of the largest channel on the max side of a comparator
109
110 for(size_t i=1; i<=INPUTS; i+=2)
111 if(m[i] > m[max])
112 max = i;
113
114 //skip over the matchings that have max matching to something smaller
115 //than max is matching to in the current matching, which is m[max - 1]
116
117 index += b*(n - 2 - m[max - 1]);
118
119 //remove m[max - 1] from the matching
120
121 for(size_t j=0; j<INPUTS; j++)
122 if(m[j] > m[max - 1])m[j]--;
123
124 //prepare for next iteration
125
126 b /= n - 3; //block size to be skipped is now smaller
127 m[max] = 0; //remove channel max from consideration
128 } //for
129
130 return index;
131} //GetIndex

139size_t CLevel2Search::Permute(CMatching& matching, const size_t n){
140 size_t nMinIndex = 99999; //minimum matching index found
141
142 for(size_t i=0; i<n; i++){
143 if(n > 2) //recurse on sub-permutation
144 nMinIndex = std::min(nMinIndex, Permute(matching, n - 1));
145
146 if(i < n - 1){ //base of recursion
147 matching.Swap(odd(n)? 0: i, n - 1); //swap to new candidate
148 nMinIndex = std::min(nMinIndex, GetIndex(matching));
149 } //if
150 } //for
151
152 return nMinIndex;
153} //Permute
