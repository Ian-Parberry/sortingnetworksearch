75size_t CSortingNetwork::FlipInput(size_t j, const size_t first, const size_t last){
76 const size_t nBit = m_nValue[first][j] ^ 1;
77 m_nZeros += nBit? -1: 1; //if nBit has flipped to 1, one less zero, else one more
78
79 for(size_t i=first; i<=last; i++){ //for each layer in range
80 m_nValue[i][j] = nBit; //flip the value on channel j at that level
81 const size_t k = m_nComparator[i][j]; //channel joined via a comparator
82 if(xor(m_nValue[i][k], j > k))
83 j = k;
84 } //for
85
86 return j;
87} //FlipInput

117bool CSortingNetwork::Sorts(){
118 size_t i = 0; //index of bit to flip
119 bool bSorts = true; //assume it sorts until we find otherwise
120 Initialize(); //intialize input and values in comparator network to zero
121
122 while(bSorts && i < m_nWidth){ //bail if it doesn't sort, or we've tried all binary inputs
123 i = m_pGrayCode->Next(); //next bit to flip in Gray code order
124 bSorts = bSorts && (i >= m_nWidth || StillSorts(i)); //check whether it still sorts when this bit is flipped
125 } //while
126
127 return bSorts;
128} //Sorts
