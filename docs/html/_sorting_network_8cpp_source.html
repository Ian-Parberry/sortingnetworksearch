66size_t CSortingNetwork::FlipInput(size_t j, const size_t first, const size_t last){
67 const size_t nBit = m_nValue[first][j] ^ 1;
68 m_nZeros += nBit? -1: 1; //if nBit has flipped to 1, one less zero, else one more
69
70 for(size_t i=first; i<=last; i++){ //for each layer in range
71 m_nValue[i][j] = nBit; //flip the value on channel j at that level
72 const size_t k = m_nComparator[i][j]; //channel joined via a comparator
73 if(xor(m_nValue[i][k], j > k))
74 j = k;
75 } //for
76
77 return j;
78} //FlipInput

112bool CSortingNetwork::Sorts(){
113 size_t i = 0; //index of bit to flip
114 bool bSorts = true; //assume it sorts until we find otherwise
115 Initialize(); //intialize input and values in comparator network to zero
116
117 while(bSorts && i < m_nWidth){ //bail if it doesn't sort, or we've tried all binary inputs
118 i = m_pGrayCode->Next(); //next bit to flip in Gray code order
119 bSorts = bSorts && (i >= m_nWidth || StillSorts(i)); //check whether it still sorts when this bit is flipped
120 } //while
121
122 return bSorts;
123} //Sorts
