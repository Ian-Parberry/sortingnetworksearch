65size_t CSortingNetwork::flipinput(size_t j, const size_t firstlayer, const size_t lastlayer){
66 for(size_t i=firstlayer; i<=lastlayer; i++){ //for each layer in range
67 m_nValue[i][j] ^= 1; //flip the value on channel j at that level
68
69 size_t k = m_nMatch[i][j]; //find the channel to which it is joined via a comparator, if any
70
71 if((m_nValue[i][k] && j>k) || !(m_nValue[i][k] || j>k))
72 j = k;
73 } //for
74
75 return j;
76} //flipinput

90bool CSortingNetwork::sorts(){
91 size_t i = 0; //index of bit to flip
92 bool bSorts = true; //assume it sorts until we find otherwise
93 initSortingTest(); //intialize input and values in comparator network to zero
94 while(bSorts && i<=INPUTS){ //bail if it doesn't sort, or we've tried all binary inputs
95 i = m_pGrayCode->next(); //next bit to flip in Gray code order
96 bSorts = bSorts && (i>INPUTS || stillsorts(i)); //check whether it still sorts when this bit is flipped
97 } //while
98 return bSorts;
99} //sorts
