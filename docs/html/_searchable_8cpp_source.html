104void CSearchable::SynchMatchingRepresentations(size_t level){
105 for(size_t j=0; j<m_nWidth; j+=2){ //for each pair of channels
106 size_t x = m_nMatching[level][j]; //channel at left end of comparator
107 size_t y = m_nMatching[level][j + 1]; //channel at the other end
108
109 if(y == m_nWidth) //if the rightmost channel is the last one in a comparator network with an odd number of inputs
110 m_nComparator[level][x] = x; //it's empty
111
112 else{ //make the testable representation
113 m_nComparator[level][x] = y; //x goes to y
114 m_nComparator[level][y] = x; //y goes to x
115 } //else
116 } //for
117} //SynchMatchingRepresentations

122void CSearchable::InitMatchingRepresentations(size_t level){
123 m_nMatching[level].Initialize(); //initialize the generatable form
124 m_nStack[level] = 0; //and its stack
125
126 for(size_t j=0; j<m_nWidth; j++) //initialize the testable form
127 m_nComparator[level][j] = j^1;
128
129 if(odd(m_nWidth)) //one extra one if n is odd
130 m_nComparator[level][m_nWidth - 1] = m_nWidth - 1;
131} //InitMatchingRepresentations

137bool CSearchable::NextComparatorNetwork(){
138 SetToS(); //set top of stack
139
140 m_nStack[m_nToS]++;
141
142 if(m_nMatching[m_nToS].Next())
143 SynchMatchingRepresentations(m_nToS);
144
145 while((m_nToS >= m_nTop) && (m_nStack[m_nToS] == m_nNumMatchings)){
146 InitMatchingRepresentations(m_nToS);
147
148 if(--m_nToS >= m_nTop){
149 m_nStack[m_nToS]++;
150
151 if(m_nStack[m_nToS] < m_nNumMatchings && m_nMatching[m_nToS].Next())
152 SynchMatchingRepresentations(m_nToS);
153 } //if
154 } //while
155
156 return m_nToS >= m_nTop; //there are no more if we blow the top of the stack
157} //NextComparatorNetwork
