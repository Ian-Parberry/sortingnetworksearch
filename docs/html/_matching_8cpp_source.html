60bool CMatching::Next(){
61 size_t s = 4;
62 size_t i = m_nStack[s - 1];
63
64 while(i < 1 && s < oddfloor(INPUTS)){
65 size_t temp = m_nMatching[s - 2];
66
67 for(size_t j=s-1; j>=2; j--){
68 m_nMatching[j - 1] = m_nMatching[j - 2];
69 m_nMap[m_nMatching[j - 1]] = j - 1;
70 } //for
71
72 m_nMatching[0] = temp;
73 m_nMap[temp] = 0;
74
75 for(size_t j=0; j<s; j++)
76 m_nStack[j] = (int)j - 1;
77
78 s += 2;
79 i = m_nStack[s - 1];
80 } //while
81
82 if(i > 0){
83 std::swap(m_nMatching[i - 1], m_nMatching[s - 2]);
84 m_nMap[m_nMatching[i - 1]] = i - 1;
85 m_nMap[m_nMatching[s - 2]] = s - 2;
86 m_nStack[s - 1] = (int)i - 1;
87 } //if
88
89 return m_nStack[i] >= 0;
90} //Next

97void SwapPair(int m[], size_t i, size_t j){
98 const size_t i0 = 2*i;
99 const size_t i1 = i0 + 1;
100 const size_t j0 = 2*j;
101 const size_t j1 = j0 + 1;
102
103 std::swap(m[i0], m[j0]);
104 std::swap(m[i1], m[j1]);
105
106 for(size_t k=0; k<INPUTS; k++)
107 if(m[k] == i0) m[k] = (int)j0;
108 else if(m[k] == (int)j0) m[k] = (int)i0;
109 else if(m[k] == (int)i1) m[k] = (int)j1;
110 else if(m[k] == (int)j1) m[k] = (int)i1;
111} //SwapPair

115void CMatching::Normalize(){
116 const size_t n = evenceil(INPUTS);
117 int m[n]; //for copy of matching, will get negative numbers during processing
118
119 for(size_t j=0; j<n; j++)
120 m[j] = (int)m_nMatching[m_nMap[j]^1];
121
122 size_t next = 1;
123
124 for(size_t j=0; j<n; j++){
125 const size_t src = std::max(next++, j/2 + 1);
126
127 if(m[j] > int(2*src + 1))
128 SwapPair(m, src, m[j]/2);
129 } //for
130
131 size_t top = 0;
132
133 for(size_t k=0; k<n; k++)
134 if(m[k] >= 0){
135 m_nMatching[top++] = k;
136 m_nMatching[top++] = m[k];
137 m[m[k]] = -1;
138 } //if
139} //Normalize

159 m_nMap[m_nMatching[i1]] = i1;
160 m_nMap[m_nMatching[j1]] = j1;
