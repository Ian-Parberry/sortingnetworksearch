42size_t CTernaryGrayCode::Next(){
43 const size_t i = m_nStack[0];
44 m_nStack[0] = 1;
45
46 const size_t j = 2*i - m_nBit[2*i - m_nDirection[i]];
47 m_nBit[j] ^= 1;
48
49 if(m_nBit[2*i] == m_nBit[2*i - 1]){
50 m_nDirection[i] ^= 1;
51 m_nStack[i - 1] = m_nStack[i];
52 m_nStack[i] = i + 1;
53 } //if
54
55 return j - 1;
56} //Next
