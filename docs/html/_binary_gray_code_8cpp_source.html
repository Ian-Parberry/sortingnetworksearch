45size_t CBinaryGrayCode::next(){
46 const size_t i = m_nStack[0]; //bit to change
47
48 m_nStack[0] = 1;
49 m_nBit[i] ^= 1; //change bit
50 m_nStack[i - 1] = m_nStack[i];
51 m_nStack[i] = i + 1;
52 m_nZeros += 1 - 2*m_nBit[i]; //adjust zero count
53
54 return i; //return bit changed
55} //next
